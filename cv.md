    function solution(input, markers) {
      return input
        .split('\n')
        .map((string) => {
          for (let i = 0; i < markers.length; i++) {
            let cutIndex = string.indexOf(markers[i]);
            if (cutIndex >= 0) {
              return string.slice(0, cutIndex).trim();
            }
          }
          return string;
        })
        .join('\n');
    };
